import { auth, signInWithEmailAndPassword } from './firebase.js'; const loginBtn = document.getElementById('loginBtn'); loginBtn.addEventListener('click', async () => { const email = document.getElementById('email').value.trim(); const password = document.getElementById('password').value.trim(); try { // Sign in with Firebase await signInWithEmailAndPassword(auth, email, password); // Redirect after success window.location.href = "admin.html"; } catch (error) { alert("Login failed: " + error.message); } });
